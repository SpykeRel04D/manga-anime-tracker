    **Step 2: Create Drizzle schema files**

    `src/db/schema/users.ts`:
    ```typescript
    import { pgTable, text, timestamp, uuid } from 'drizzle-orm/pg-core'

    export const users = pgTable('users', {
      id: uuid('id').defaultRandom().primaryKey(),
      email: text('email').notNull().unique(),
      passwordHash: text('password_hash').notNull(),
      name: text('name'),
      createdAt: timestamp('created_at', { withTimezone: true }).defaultNow().notNull(),
      updatedAt: timestamp('updated_at', { withTimezone: true }).defaultNow().notNull(),
    })
    ```

    `src/db/schema/tracking.ts`:
    ```typescript
    import { integer, pgEnum, pgTable, text, timestamp, uuid } from 'drizzle-orm/pg-core'

    import { users } from './users'

    export const trackingStatusEnum = pgEnum('tracking_status', [
      'watching',
      'completed',
      'on_hold',
      'dropped',
      'plan_to_watch',
    ])

    export const mediaTypeEnum = pgEnum('media_type', ['anime', 'manga'])

    export const trackingEntries = pgTable('tracking_entries', {
      id: uuid('id').defaultRandom().primaryKey(),
      userId: uuid('user_id').notNull().references(() => users.id, { onDelete: 'cascade' }),
      anilistId: integer('anilist_id').notNull(),
      mediaType: mediaTypeEnum('media_type').notNull(),
      status: trackingStatusEnum('status').notNull().default('plan_to_watch'),
      progress: integer('progress').notNull().default(0),
      rating: integer('rating'),
      notes: text('notes'),
      titleEnglish: text('title_english'),
      titleRomaji: text('title_romaji'),
      coverImageUrl: text('cover_image_url'),
      totalEpisodes: integer('total_episodes'),
      totalChapters: integer('total_chapters'),
      createdAt: timestamp('created_at', { withTimezone: true }).defaultNow().notNull(),
      updatedAt: timestamp('updated_at', { withTimezone: true }).defaultNow().notNull(),
    })
    ```

    `src/db/schema/index.ts`:
    ```typescript
    export { users } from './users'
    export { trackingEntries, trackingStatusEnum, mediaTypeEnum } from './tracking'
    ```

    **Step 3: Create environment-aware DB connection**

    `src/db/drizzle.ts`:
    - Import `drizzle` from `drizzle-orm/neon-http` for production (Neon)
    - Import `drizzle` from `drizzle-orm/node-postgres` for local (Docker PostgreSQL)
    - Check `NODE_ENV === 'production'` to select driver
    - Use `NEON_DATABASE_URL` for production, `LOCAL_DATABASE_URL` for local
    - Import and pass `* as schema` from `./schema`
    - Export `db` singleton
    - See 01-RESEARCH.md "Pattern 2: Environment-Aware Database Connection" for exact pattern

    **Step 4: Create Drizzle Kit config**

    `drizzle.config.ts`:
    - Import `dotenv/config`
    - Schema path: `./src/db/schema/index.ts`
    - Output path: `./src/db/migrations`
    - Dialect: `postgresql`
    - URL: Use `NEON_DATABASE_URL` if `NODE_ENV=production`, else `LOCAL_DATABASE_URL`
    - See 01-RESEARCH.md "Drizzle Config" code example

    **Step 5: Create shared utilities**

    `src/lib/utils.ts` (shadcn/ui cn helper):
    ```typescript
    import { type ClassValue, clsx } from 'clsx'
    import { twMerge } from 'tailwind-merge'

    export function cn(...inputs: ClassValue[]): string {
      return twMerge(clsx(inputs))
    }
    ```

    **Step 6: Push schema to local DB**
